// Fragment shader
#version 330 core
uniform sampler2D textureSampler;
uniform vec3 baseColor; // Base color of the plane

in vec2 TexCoords;
out vec4 FragColor;

void main() {
    vec4 texColor = texture(textureSampler, TexCoords);
    // Use the texture color and base color without lighting
    FragColor = texColor * vec4(baseColor, 1.0);
}
